	* tabbarItem方法：
		>  
		>  /*swiper切换*/  
		  swiperChange: function (e) {  
		    var that = this;
		    that.setData({ currentTab: e.detail.current });  
		  },  
		  /*点击tab切换*/  
		  tabbarItem: function (e) {  
		    var that = this;  
		    if (this.data.currentTab === e.target.dataset.current) {  
		      return false;  
		    } else {  
		      that.setData({  
		        currentTab: e.target.dataset.current  
		      })  
		    }  
		  }
